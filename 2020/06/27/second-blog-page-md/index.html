hexo博客搭建
npm install -g cnpm –registry=https://registry.npm.taobao.org # 从淘宝下载中国的包管理器，这样速度更快
cnpm install -g hexo-cli

hexo s # 启动博客
hexo n # 写篇文章

hexo clean
hexo g # 生成
hexo d # 部署
cnpm install –save hexo-deployer-git # 将博客部署到GitHub
